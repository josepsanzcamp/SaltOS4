
# we are creating colorized themes derived from cosmo

cd build

npm install bootstrap bootswatch

rm -f ../cosmo.*.min.css

for i in blue purple orange green cyan pink gray slate crimson teal amber indigo; do
    sass cosmo.$i.scss ../cosmo.$i.min.css --style=compressed --no-source-map
done

rm -rf node_modules
rm -f package*.json

cd ..
