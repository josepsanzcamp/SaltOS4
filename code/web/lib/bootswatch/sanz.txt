
# important note: this themes have been modified to:
# - remove the @import of external fonts
# - add or fixe some issues with the SaltOS4 UI

themes=$(curl -s https://bootswatch.com/api/5.json | jq -r .themes[].name | tr '[:upper:]' '[:lower:]')
for i in $themes; do
    wget -O $i.min.css https://cdn.jsdelivr.net/npm/bootswatch@5.3.5/dist/$i/bootstrap.min.css
    cat $i.min.css | grep -v sourceMappingURL > $i.min.css.new
    mv $i.min.css.new $i.min.css
done

php ../../../../scripts/make_bootswatch.php
